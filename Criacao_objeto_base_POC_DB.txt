CREATE DATABASE POC_DB;

CREATE TABLE POC_DB.CLIENTE_PF 
(
	CD_CLIENTE BIGINT, 
	NR_CPF VARCHAR (12), 
	NM_NOME VARCHAR (100), 
	NR_IDADE INT, 
	DT_NASCIMENTO DATE,
	DS_ETNIA VARCHAR (9),
	DS_RELIGIAO VARCHAR (20), 
	DS_SEXO CHAR(1), 
	DS_ESTADO_CIVIL VARCHAR (10),
	DS_NACIONALIDADE VARCHAR (100),
	DS_NATURALIDADE VARCHAR (100),
	DS_LOGRADOURO VARCHAR (100),
	DS_CIDADE VARCHAR (100),
	DS_UF CHAR (2),
	NR_CEP VARCHAR (9)
)
ROW FORMAT DELIMITED FIELDS TERMINATED BY ','
STORED AS ORC;


CREATE TABLE POC_DB.FATURAS_PF
(
	CD_FATURA BIGINT, 
	CD_CLIENTE BIGINT,  
	CD_UND_CONSUMIDORA VARCHAR (14)
	DT_VENCIMENTO DATE,
	DT_PAGAMENTO DATE,
	DT_LEITURA DATE,
	VR_FATURA DECIMAL (9,2),
	QT_CONSUMO_KWh DECIMAL (9,2)	
)
ROW FORMAT DELIMITED FIELDS TERMINATED BY ','
STORED AS ORC;
